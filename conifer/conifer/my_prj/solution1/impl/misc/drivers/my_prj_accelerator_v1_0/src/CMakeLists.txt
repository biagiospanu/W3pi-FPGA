cmake_minimum_required(VERSION 3.15)
project(my_prj_accelerator)
find_package(common)
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
collect (PROJECT_LIB_SOURCES xmy_prj_accelerator.c)
collect (PROJECT_LIB_SOURCES xmy_prj_accelerator_sinit.c)
collect (PROJECT_LIB_SOURCES xmy_prj_accelerator_linux.c)
collect (PROJECT_LIB_SOURCES xmy_prj_accelerator_g.c)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collect (PROJECT_LIB_HEADERS xmy_prj_accelerator.h)
collect (PROJECT_LIB_HEADERS xmy_prj_accelerator_hw.h)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
include_directories(${CMAKE_BINARY_DIR}/include)
add_library(my_prj_accelerator STATIC ${_sources})
set_target_properties(my_prj_accelerator PROPERTIES LINKER_LANGUAGE C)
